class Queue :<br>
    def __init__(self) :<br>
        self.items = []<br>
    def enqueue(self, item) :<br>
        self.items.insert(0,item)<br>
    def dequeue(self) :<br>
        return self.items.pop()<br>
    def peek(self) :<br>
        return self.items[0]<br>
    def isEmpty(self) :<br>
        return self.items == []<br>
    def size(self) :<br>
        return len(self.items) # Queue 정의<br>
class lens :<br>
    def __init__(self, tp, focus, l): # 변수 초기화<br>
        self.tp = tp<br>

M = round(m,2)
